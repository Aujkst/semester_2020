  1. 简单回归
library(MASS)
lmp <- function(x) {
    lm <- lm(Boston$crim ~ x)
    f <- summary(lm)$fstatistic
    p <- pf(f[1],f[2],f[3], lower.tail=F)
    names(p) <- NULL
    return(p)
}

results <- apply(Boston[, -1], 2, lmp)
names(results) <- paste(names(Boston)[1], 
                        names(Boston)[2:14], 
                        sep = " ~ ")
results[results>=0.00001]
## crim ~ chas 
##   0.2094345

由结果可以看出，除了chas以外，其他预测变量与人均犯罪率crim的模型都非常显著（\(P-Value<0.00001\)）。

    lm <- lm(Boston$crim ~ x+I(x^2)+I(x^3))
    cf <- summary(lm)$coefficients
    return(cf[-1,4])
